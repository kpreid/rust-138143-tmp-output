6/// Pick a name for a type generated by this macro.
7/// Such names can conflict with the macro caller’s own types, so we want them to be named in
8/// a predictable fashion.
9///
10/// `input_type_name` is the name of the type our derive macro was applied to.
11/// `role_name` is a camel-case component describing this type in particular, e.g. “Iter”.
12///
13/// This function is also available as a method on `ExhaustContext`.
14pub fn generated_type_name(input_type_name: &str, role_name: &str) -> Ident {
15    Ident::new(
16        &format!("Exhaust{input_type_name}{role_name}"),
17        Span::mixed_site(),
18    )
19}

21/// Data and helpers for generating an exhaustive iterator, that are common
22/// between the enum and struct versions.
23pub(crate) struct ExhaustContext {
24    /// Maximum visibility for the generated types, inherited from the declaration
25    /// (e.g. `pub Foo` should have a `pub ExhaustFoo` iterator).
26    pub vis: syn::Visibility,
27
28    /// Generics present on the declaration, which need to be copied to the
29    /// iterator.
30    pub generics: syn::Generics,
31
32    /// Name of the type being exhausted, which is also the type the macro is applied to.
33    pub item_type: ConstructorSyntax,
34
35    /// Name of the generated factory type, which is a unit struct containing a struct
36    /// like the type being exhausted but with different field types,
37    /// and is the item type of the generated iterator.
38    pub factory_type: ConstructorSyntax,
39
40    /// Name of the generated iterator type.
41    pub iterator_type_name: Ident,
42
43    /// Path by which the `exhaust` crate should be referred to.
44    pub exhaust_crate_path: syn::Path,
45}

47impl ExhaustContext {
48    /// Pick a name for a type generated by this macro.
49    /// Such names can conflict with the macro caller’s own types, so we want them to be named in
50    /// a predictable fashion.
51    ///
52    /// `input_type_name` is the name of the type our derive macro was applied to.
53    /// `role_name` is a camel-case component describing this type in particular, e.g. “Iter”.
54    ///
55    /// This method is also available as a free function to work without a context.
56    pub fn generated_type_name(&self, role_name: &str) -> Result<Ident, syn::Error> {
57        Ok(generated_type_name(
58            &self.item_type.name_for_incorporation()?,
59            role_name,
60        ))
61    }

63    /// Generate the TraitBound containing the `exhaust::Exhaust` trait.
64    pub fn exhaust_trait_bound(&self) -> syn::TraitBound {
65        let mut path = self.exhaust_crate_path.clone();
66        path.segments.push(parse_quote! { Exhaust });
67        // reinterpret as TraitBound
68        parse_quote! { #path }
69    }

71    /// As [`syn::Generics::split_for_impl`], but adding the given bounds,
72    /// as well as the `::exhaust::Exhaust` bound.
73    pub fn generics_with_bounds(
74        &self,
75        mut bounds: Punctuated<syn::TypeParamBound, syn::Token![+]>,
76    ) -> (
77        syn::ImplGenerics<'_>,
78        syn::TypeGenerics<'_>,
79        Punctuated<syn::WherePredicate, syn::token::Comma>,
80    ) {
81        bounds.push(syn::TypeParamBound::Trait(self.exhaust_trait_bound()));
82        let generics = &self.generics;
83        let (impl_generics, ty_generics, where_clause) = generics.split_for_impl();
84        let mut augmented_where_predicates = match where_clause {
85            Some(clause) => clause.predicates.clone(),
86            None => Punctuated::new(),
87        };
88        for g in generics.params.iter() {
89            if let syn::GenericParam::Type(g) = g {
90                augmented_where_predicates.push(syn::WherePredicate::Type(syn::PredicateType {
91                    lifetimes: None,
92                    bounded_ty: syn::Type::Verbatim(g.ident.to_token_stream()),
93                    colon_token: <_>::default(),
94                    bounds: bounds.clone(),
95                }));
96            }
97        }
98        (impl_generics, ty_generics, augmented_where_predicates)
99    }

101    pub fn iterator_doc(&self) -> String {
102        match &self.item_type {
103            // TODO: No tests to validate this doc link
104            ConstructorSyntax::Braced(name) => format!(
105                "Iterator over all values of [`{name}`].\n\
106                \n\
107                To obtain an instance of this iterator, call [`Exhaust::exhaust()`].\n\
108                \n\
109                [`Exhaust::exhaust()`]: {crate}::Exhaust",
110                crate = self.exhaust_crate_path.to_token_stream(),
111            ),
112            ConstructorSyntax::Tuple => {
113                format!(
114                    "Iterator over all tuples of {len} elements.\n\
115                    \n\
116                    To obtain an instance of this iterator, call [`Exhaust::exhaust()`].",
117                    len = self.generics.params.len(),
118                )
119            }
120        }
121    }

123    /// Generate the common parts of the Iterator and factory implementation.
124    pub fn impl_iterator_and_factory_traits(
125        &self,
126        iterator_next_body: TokenStream2,
127        iterator_default_body: TokenStream2,
128        iterator_clone_body: TokenStream2,
129    ) -> TokenStream2 {
130        let exhaust_crate_path = &self.exhaust_crate_path;
131        let iterator_type_name = &self.iterator_type_name;
132        let (impl_generics, ty_generics, augmented_where_predicates) =
133            self.generics_with_bounds(syn::parse_quote! {});
134        let item_type_inst = self.item_type.parameterized(&self.generics);
135
136        let factory_impls = match &self.factory_type {
137            // Don't try to implement for a tuple type.
138            ConstructorSyntax::Tuple => quote! {},
139
140            ConstructorSyntax::Braced(factory_outer_type_name) => quote! {
141                impl #impl_generics ::core::fmt::Debug for #factory_outer_type_name #ty_generics
142                where #augmented_where_predicates {
143                    fn fmt(&self, f: &mut ::core::fmt::Formatter<'_>) -> ::core::fmt::Result {
144                        // TODO: Print the contents of the factory.
145                        // We know this is possible because the `Exhaust` trait requires
146                        // `Self::Factory: Debug`.
147                        //
148                        // Ideally we would just print `T::from_factory(self)`, but we don't know
149                        // whether that implements `Debug` or not.
150                        // (But a macro helper attribute could let the user tell us that.)
151                        f.debug_struct(stringify!(#factory_outer_type_name))
152                            .finish_non_exhaustive()
153                    }
154                }
155            },
156        };
157
158        quote! {
159            impl #impl_generics ::core::iter::Iterator for #iterator_type_name #ty_generics
160            where #augmented_where_predicates {
161                type Item = <#item_type_inst as #exhaust_crate_path::Exhaust>::Factory;
162
163                fn next(&mut self) -> ::core::option::Option<Self::Item> {
164                    #iterator_next_body
165                }
166            }
167
168            impl #impl_generics ::core::iter::FusedIterator for #iterator_type_name #ty_generics
169            where #augmented_where_predicates {}
170
171            impl #impl_generics ::core::default::Default for #iterator_type_name #ty_generics
172            where #augmented_where_predicates {
173                fn default() -> Self {
174                    #iterator_default_body
175                }
176            }
177
178            // A manual impl of Debug is required to provide the right bounds on the generics.
179            impl #impl_generics ::core::fmt::Debug for #iterator_type_name #ty_generics
180            where #augmented_where_predicates {
181                fn fmt(&self, f: &mut ::core::fmt::Formatter<'_>) -> ::core::fmt::Result {
182                    // TODO: Print the current enum state.
183                    // We know this is possible because the `Exhaust` trait requires
184                    // `Self::Iter: Debug`.
185                    f.debug_struct(stringify!(#iterator_type_name))
186                        .finish_non_exhaustive()
187                }
188            }
189
190            // A manual impl of Clone is required to *not* have a `Clone` bound on the generics.
191            impl #impl_generics ::core::clone::Clone for #iterator_type_name #ty_generics
192            where #augmented_where_predicates {
193                fn clone(&self) -> Self {
194                    #iterator_clone_body
195                }
196            }
197
198            #factory_impls
199        }
200    }

203/// How to name a type for construction.
204pub(crate) enum ConstructorSyntax {
205    /// A struct or variant name to used with `MyStruct { field: value }` syntax.
206    Braced(TokenStream2),
207    /// The type is a primitive tuple.
208    Tuple,
209}
210
211impl ConstructorSyntax {
212    /// Name to use for concatenation to construct new type names.
213    pub fn name_for_incorporation(&self) -> Result<String, syn::Error> {
214        match self {
215            ConstructorSyntax::Braced(name) => Ok(name.to_string()),
216            // we could return “Tuple” here if it ever became needed.
217            ConstructorSyntax::Tuple => Err(syn::Error::new(
218                Span::call_site(),
219                "exhaust-macros internal error: no name for tuple types",
220            )),
221        }
222    }
223
224    /// Returns the path for use in a type declaration or pattern.
225    pub(crate) fn path(&self) -> Result<&TokenStream2, syn::Error> {
226        match self {
227            ConstructorSyntax::Braced(name) => Ok(name),
228            ConstructorSyntax::Tuple => Err(syn::Error::new(
229                Span::call_site(),
230                "exhaust-macros internal error: no name for tuple types",
231            )),
232        }
233    }
234
235    /// Type applied to given type parameters.
236    pub(crate) fn parameterized(&self, generics: &syn::Generics) -> TokenStream2 {
237        match self {
238            ConstructorSyntax::Braced(name) => {
239                let (_, ty_generics, _) = generics.split_for_impl();
240                quote! { #name #ty_generics }
241            }
242            ConstructorSyntax::Tuple => {
243                let par = generics.type_params();
244                quote! { ( #( #par , )* ) }
245            }
246        }
247    }
248
249    /// Constructor applied to fields.
250    /// The fields MUST be in original declardd order, to handle the tuple case.
251    pub(crate) fn value_expr<'a>(
252        &self,
253        names: impl Iterator<Item = &'a TokenStream2>,
254        values: impl Iterator<Item = &'a TokenStream2>,
255    ) -> TokenStream2 {
256        match self {
257            ConstructorSyntax::Braced(name) => {
258                quote! { #name { #( #names : #values , )* } }
259            }
260            ConstructorSyntax::Tuple => {
261                quote! { ( #( #values , )* ) }
262            }
263        }
264    }
265
266    /// Given an enum type name, produce a variant constructor.
267    pub(crate) fn with_variant(&self, target_variant_ident: &Ident) -> ConstructorSyntax {
268        match self {
269            ConstructorSyntax::Braced(name) => {
270                let mut name = name.clone();
271                name.extend(quote! { :: });
272                name.extend(target_variant_ident.to_token_stream());
273                ConstructorSyntax::Braced(name)
274            }
275            ConstructorSyntax::Tuple => panic!("a tuple is not an enum"),
276        }
277    }
278}

280/// Generate arms for a match which maps every field of every variant of an enum.
281pub(crate) fn clone_like_match_arms(
282    variants: &Punctuated<syn::Variant, syn::Token![,]>,
283    input_type_name: &TokenStream2,
284    output_type_name: &TokenStream2,
285    binding_mode: &TokenStream2,
286    value_transform: impl Fn(TokenStream2) -> TokenStream2,
287) -> Vec<TokenStream2> {
288    variants
289        .iter()
290        .map(|target_variant| {
291            let variant_name = &target_variant.ident;
292            match &target_variant.fields {
293                syn::Fields::Named(_) => {
294                    let members = target_variant.fields.members().collect::<Vec<_>>();
295                    let values = members
296                        .iter()
297                        .map(|member| value_transform(member.to_token_stream()))
298                        .collect::<Vec<_>>();
299                    quote! {
300                        #input_type_name::#variant_name {
301                            #( #binding_mode #members, )*
302                        } => #output_type_name::#variant_name {
303                            #( #members: #values, )*
304                        }
305                    }
306                }
307                syn::Fields::Unnamed(fields) => {
308                    let vars: Vec<Ident> = (0..fields.unnamed.len())
309                        .map(|i| Ident::new(&format!("clone_f{i}"), Span::mixed_site()))
310                        .collect();
311                    let values = vars
312                        .iter()
313                        .map(|var| value_transform(var.to_token_stream()))
314                        .collect::<Vec<_>>();
315                    quote! {
316                        #input_type_name::#variant_name(
317                            #( #binding_mode #vars, )*
318                        ) => #output_type_name::#variant_name(
319                            #( #values, )*
320                        )
321                    }
322                }
323                syn::Fields::Unit => quote! {
324                    #input_type_name::#variant_name => #output_type_name::#variant_name
325                },
326            }
327        })
328        .collect()
329}

331/// Generate a match arm which is a transformation which maps every field of the struct.
332pub(crate) fn clone_like_struct_conversion(
333    fields: &syn::Fields,
334    input_type_name: &TokenStream2,
335    output_type_name: &TokenStream2,
336    binding_mode: &TokenStream2,
337    value_transform: impl Fn(TokenStream2) -> TokenStream2,
338) -> TokenStream2 {
339    match fields {
340        syn::Fields::Named(_) => {
341            let members = fields.members().collect::<Vec<_>>();
342            let values = members
343                .iter()
344                .map(|member| value_transform(member.to_token_stream()))
345                .collect::<Vec<_>>();
346            quote! {
347                #input_type_name {
348                    #( #binding_mode #members, )*
349                } => #output_type_name {
350                    #( #members: #values, )*
351                }
352            }
353        }
354        syn::Fields::Unnamed(fields) => {
355            let vars: Vec<Ident> = (0..fields.unnamed.len())
356                .map(|i| Ident::new(&format!("clone_f{i}"), Span::mixed_site()))
357                .collect();
358            let values = vars
359                .iter()
360                .map(|var| value_transform(var.to_token_stream()))
361                .collect::<Vec<_>>();
362            quote! {
363                #input_type_name(
364                    #( #binding_mode #vars, )*
365                ) => #output_type_name(
366                    #( #values, )*
367                )
368            }
369        }
370        syn::Fields::Unit => quote! {
371            #input_type_name => #output_type_name
372        },
373    }
